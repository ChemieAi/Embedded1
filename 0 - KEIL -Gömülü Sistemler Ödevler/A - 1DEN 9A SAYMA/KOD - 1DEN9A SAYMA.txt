#include "TM4C123.h" // Device header
uint8_t const SEG_Tablo[16]={0x3F, 0x06,0x5B,0x4F,0x66,0x6D,0x7D,0x07,0x7F,0x6F,0x77,0x7C,0x39,0x5E,0x79,0x71};

void ms_delay(int32_t ms);

int main() {
		
		uint8_t say=0;
    SYSCTL->RCGC2 |= 0x00000022; // Enabling clock for PORTF  AND PORTB
    GPIOF->DIR |= 0X02;          // Set PIN 1 (Red) as output
    GPIOF->DEN |= 0X02;          // Set PIN 1 (Red) as digital
		GPIOB->DIR |= 0XFF;						
	  GPIOB->DEN |= 0XFF;
	
    while (1) {
			
				for(say=0;say<10;say++)
        {
					GPIOB->DATA = SEG_Tablo[say];
					ms_delay(500);
        }
        

		}
}

void ms_delay(int32_t ms) {
    for (int i = 0; i < ms; i++) {
        for (int j = 0; j < 3180; j++) {
            __NOP();
        }
    }
}
